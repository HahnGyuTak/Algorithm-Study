#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

void swap(int** a, int** b)
{
	int* t = *a;
	*a = *b;
	*b = t;
}

void state_heap(int** arr, int end)
{
	for (int i = 1; i < end; i++)
	{
		int child = i;
		do
		{
			int parent = (child - 1) / 2;
			if (arr[child][0] > arr[parent][0]) swap(arr + child, arr + parent);
			else if (arr[child][0] == arr[parent][0])
				if (arr[child][1] > arr[parent][1])
					swap(arr + child, arr + parent);
			child = parent;
		} while (child != 0);
	}
}

void heap_sorting(int** ar, int size)
{
	state_heap(ar, size);
	for (int i = size - 1; i >= 0; i--)
	{
		{ int* t; t = ar[0]; ar[0] = ar[i]; ar[i] = t; }
		int parent = 0;
		int child = 1;
		do
		{
			child = 2 * parent + 1;
			if (child < i - 1 && ar[child][0] < ar[child + 1][0]) child++;
			else if (child < i - 1 && ar[child][0] == ar[child + 1][0])
				if (ar[child][1] < ar[child + 1][1])
					child++;

			if (child < i && ar[parent][0] < ar[child][0])
				swap(ar + child, ar + parent);
			else if (child < i && ar[child][0] == ar[parent][0])
				if (ar[child][1] > ar[parent][1])
					swap(ar + child, ar + parent);
			parent = child;
		} while (child < i);
	}
}

int main() {

	int N, ** ar = NULL;

	scanf("%d", &N);
	ar = (int*)calloc(N, sizeof(int*));
	for (int i = 0; i < N; i++)
	{
		ar[i] = (int)calloc(2, sizeof(int));
		scanf("%d %d", ar[i], ar[i] + 1);
	}
	heap_sorting(ar, N);
	for (int i = 0; i < N; i++)
		printf("%d %d\n", ar[i][0], ar[i][1]);
}
